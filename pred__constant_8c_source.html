73 pred_constant_update(const struct XCSF *xcsf, const struct Cl *c,
74  const double *x, const double *y)
75 {
76  (void) x;
77  if (c->exp * xcsf->BETA < 1) {
78  for (int var = 0; var < xcsf->y_dim; ++var) {
79  c->prediction[var] =
80  (c->prediction[var] * (c->exp - 1) + y[var]) / c->exp;
81  }
82  } else {
83  for (int var = 0; var < xcsf->y_dim; ++var) {
84  c->prediction[var] += xcsf->BETA * (y[var] - c->prediction[var]);
85  }
86  }
87 }
