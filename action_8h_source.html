266 static inline void
267 act_json_import(const struct XCSF *xcsf, struct Cl *c, const cJSON *json)
268 {
269  const cJSON *item = cJSON_GetObjectItem(json, "type");
270  if (item == NULL || !cJSON_IsString(item)) {
271  printf("action_json_import(): missing type\n");
272  exit(EXIT_FAILURE);
273  }
274  const char *type = item->valuestring;
275  if (action_type_as_int(type) != xcsf->act->type) {
276  printf("action_json_import(): mismatched type\n");
277  printf("XCSF type = %s, but imported type = %s\n",
278  action_type_as_string(xcsf->act->type), type);
279  exit(EXIT_FAILURE);
280  }
281  (*c->act_vptr->act_impl_json_import)(xcsf, c, json);
282 }
