39 void
40 condition_set(const struct XCSF *xcsf, struct Cl *c)
41 {
42  switch (xcsf->cond->type) {
43  case COND_TYPE_DUMMY:
44  c->cond_vptr = &cond_dummy_vtbl;
45  break;
46  case COND_TYPE_HYPERRECTANGLE:
47  c->cond_vptr = &cond_rectangle_vtbl;
48  break;
49  case COND_TYPE_HYPERELLIPSOID:
50  c->cond_vptr = &cond_ellipsoid_vtbl;
51  break;
52  case COND_TYPE_NEURAL:
53  c->cond_vptr = &cond_neural_vtbl;
54  break;
55  case COND_TYPE_GP:
56  c->cond_vptr = &cond_gp_vtbl;
57  break;
58  case COND_TYPE_DGP:
59  c->cond_vptr = &cond_dgp_vtbl;
60  break;
61  case COND_TYPE_TERNARY:
62  c->cond_vptr = &cond_ternary_vtbl;
63  break;
64  case RULE_TYPE_DGP:
65  c->cond_vptr = &rule_dgp_cond_vtbl;
66  c->act_vptr = &rule_dgp_act_vtbl;
67  break;
68  case RULE_TYPE_NEURAL:
69  c->cond_vptr = &rule_neural_cond_vtbl;
70  c->act_vptr = &rule_neural_act_vtbl;
71  break;
72  default:
73  printf("Invalid condition type specified: %d\n", xcsf->cond->type);
74  exit(EXIT_FAILURE);
75  }
76 }

195 static void
196 cond_param_defaults_neural(struct XCSF *xcsf)
197 {
198  // hidden layer
199  struct ArgsLayer *args = malloc(sizeof(struct ArgsLayer));
200  layer_args_init(args);
201  args->type = CONNECTED;
202  args->n_inputs = xcsf->x_dim;
203  args->n_init = 10;
204  args->n_max = 100;
205  args->max_neuron_grow = 1;
206  args->function = LOGISTIC;
207  args->evolve_weights = true;
208  args->evolve_neurons = true;
209  args->evolve_connect = true;
210  xcsf->cond->largs = args;
211  // output layer
212  args->next = layer_args_copy(args);
213  args->next->function = LINEAR;
214  args->next->n_inputs = args->n_init;
215  args->next->n_max = 1;
216  args->next->evolve_neurons = false;
217 }

350 {

352  size_t s = 0;

355  s += fwrite(&cond->min, sizeof(double), 1, fp);
356  s += fwrite(&cond->max, sizeof(double), 1, fp);

358  s += fwrite(&cond->p_dontcare, sizeof(double), 1, fp);
359  s += fwrite(&cond->bits, sizeof(int), 1, fp);

361  s += tree_args_save(cond->targs, fp);
