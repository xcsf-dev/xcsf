40 void
41 condition_set(const struct XCSF *xcsf, struct Cl *c)
42 {
43  switch (xcsf->cond->type) {
44  case COND_TYPE_DUMMY:
45  c->cond_vptr = &cond_dummy_vtbl;
46  break;
47  case COND_TYPE_HYPERRECTANGLE:
48  c->cond_vptr = &cond_rectangle_vtbl;
49  break;
50  case COND_TYPE_HYPERELLIPSOID:
51  c->cond_vptr = &cond_ellipsoid_vtbl;
52  break;
53  case COND_TYPE_NEURAL:
54  c->cond_vptr = &cond_neural_vtbl;
55  break;
56  case COND_TYPE_GP:
57  c->cond_vptr = &cond_gp_vtbl;
58  break;
59  case COND_TYPE_DGP:
60  c->cond_vptr = &cond_dgp_vtbl;
61  break;
62  case COND_TYPE_TERNARY:
63  c->cond_vptr = &cond_ternary_vtbl;
64  break;
65  case RULE_TYPE_DGP:
66  c->cond_vptr = &rule_dgp_cond_vtbl;
67  c->act_vptr = &rule_dgp_act_vtbl;
68  break;
69  case RULE_TYPE_NEURAL:
70  c->cond_vptr = &rule_neural_cond_vtbl;
71  c->act_vptr = &rule_neural_act_vtbl;
72  break;
73  default:
74  printf("Invalid condition type specified: %d\n", xcsf->cond->type);
75  exit(EXIT_FAILURE);
76  }
77 }

280 const char *
281 cond_param_json_export(const struct XCSF *xcsf)
282 {
283  const struct ArgsCond *cond = xcsf->cond;
284  cJSON *json = cJSON_CreateObject();
285  cJSON_AddStringToObject(json, "type", condition_type_as_string(cond->type));
286  cJSON *params = NULL;
287  switch (cond->type) {
288  case COND_TYPE_TERNARY:
289  params = cJSON_Parse(cond_param_json_export_ternary(xcsf));
290  break;
291  case COND_TYPE_HYPERELLIPSOID:
292  case COND_TYPE_HYPERRECTANGLE:
293  params = cJSON_Parse(cond_param_json_export_csr(xcsf));
294  break;
295  case COND_TYPE_GP:
296  params = cJSON_Parse(tree_args_json_export(xcsf->cond->targs));
297  break;
298  case COND_TYPE_DGP:
299  case RULE_TYPE_DGP:
300  params = cJSON_Parse(graph_args_json_export(xcsf->cond->dargs));
301  break;
302  case COND_TYPE_NEURAL:
303  case RULE_TYPE_NEURAL:
304  case RULE_TYPE_NETWORK:
305  params = cJSON_Parse(layer_args_json_export(xcsf->cond->largs));
306  break;
307  default:
308  break;
309  }
310  if (params != NULL) {
311  cJSON_AddItemToObject(json, "args", params);
312  }
313  const char *string = cJSON_Print(json);
314  cJSON_Delete(json);
315  return string;
316 }

324 size_t
325 cond_param_save(const struct XCSF *xcsf, FILE *fp)
326 {
327  const struct ArgsCond *cond = xcsf->cond;
328  size_t s = 0;
329  s += fwrite(&cond->type, sizeof(int), 1, fp);
330  s += fwrite(&cond->eta, sizeof(double), 1, fp);
331  s += fwrite(&cond->min, sizeof(double), 1, fp);
332  s += fwrite(&cond->max, sizeof(double), 1, fp);
333  s += fwrite(&cond->spread_min, sizeof(double), 1, fp);
334  s += fwrite(&cond->p_dontcare, sizeof(double), 1, fp);
335  s += fwrite(&cond->bits, sizeof(int), 1, fp);
336  s += graph_args_save(cond->dargs, fp);
337  s += tree_args_save(cond->targs, fp);
338  s += layer_args_save(cond->largs, fp);
339  return s;
