146 double
147 cl_del_vote(const struct XCSF *xcsf, const struct Cl *c, const double avg_fit)
148 {
149  if (c->exp > xcsf->THETA_DEL && c->fit < xcsf->DELTA * avg_fit * c->num) {
150  return c->size * c->num * avg_fit / (c->fit / c->num);
151  }
152  return c->size * c->num;
153 }

161 double
162 cl_acc(const struct XCSF *xcsf, const struct Cl *c)
163 {
164  if (c->err > xcsf->E0) {
165  const double acc = xcsf->ALPHA * pow(c->err / xcsf->E0, -(xcsf->NU));
166  return fmax(acc, DBL_EPSILON);
167  }
168  return 1;

182 void
183 cl_update(const struct XCSF *xcsf, struct Cl *c, const double *x,
184  const double *y, const int set_num, const bool cur)
185 {
186  ++(c->exp);
187  if (!cur) { // propagate inputs for the previous state update
188  cl_predict(xcsf, c, x);
189  }
190  const double error = (xcsf->loss_ptr)(xcsf, c->prediction, y);
191  if (c->exp * xcsf->BETA < 1) {
192  c->err = (c->err * (c->exp - 1) + error) / c->exp;
193  c->size = (c->size * (c->exp - 1) + set_num) / c->exp;
194  } else {
195  c->err += xcsf->BETA * (error - c->err);
196  c->size += xcsf->BETA * (set_num - c->size);
197  }
198  cond_update(xcsf, c, x, y);
199  pred_update(xcsf, c, x, y);
200  act_update(xcsf, c, x, y);
201 }

483 const char *
484 cl_json_export(const struct XCSF *xcsf, const struct Cl *c,
485  const bool return_cond, const bool return_act,
486  const bool return_pred)
487 {
488  cJSON *json = cJSON_CreateObject();

490  cJSON_AddNumberToObject(json, "fitness", c->fit);
491  cJSON_AddNumberToObject(json, "accuracy", cl_acc(xcsf, c));
492  cJSON_AddNumberToObject(json, "set_size", c->size);
493  cJSON_AddNumberToObject(json, "numerosity", c->num);
494  cJSON_AddNumberToObject(json, "experience", c->exp);
495  cJSON_AddNumberToObject(json, "time", c->time);
496  cJSON_AddNumberToObject(json, "samples_seen", c->age);
497  cJSON_AddNumberToObject(json, "samples_matched", c->mtotal);
498  cJSON_AddBoolToObject(json, "current_match", c->m);
499  cJSON_AddNumberToObject(json, "current_action", c->action);
500  cJSON *p = cJSON_CreateDoubleArray(c->prediction, xcsf->y_dim);
501  cJSON_AddItemToObject(json, "current_prediction", p);
502  if (return_cond) {
503  cJSON *condition = cJSON_Parse(cond_json_export(xcsf, c));
504  cJSON_AddItemToObject(json, "condition", condition);
505  }
506  if (return_act) {
507  cJSON *action = cJSON_Parse(act_json_export(xcsf, c));
508  cJSON_AddItemToObject(json, "action", action);
509  }
510  if (return_pred) {
511  cJSON *prediction = cJSON_Parse(pred_json_export(xcsf, c));
512  cJSON_AddItemToObject(json, "prediction", prediction);
513  }
514  const char *string = cJSON_Print(json);
515  cJSON_Delete(json);
516  return string;
517 }
