189 static bool
190 mutate_neurons(struct Layer *l)
191 {

193  if (n != 0) {

197  layer_add_neurons(l->uo, n);

199  layer_add_neurons(l->wi, n);
200  layer_add_neurons(l->wg, n);
201  layer_add_neurons(l->wo, n);
202  layer_resize(l->wf, l->uf);
203  layer_resize(l->wi, l->uf);
204  layer_resize(l->wg, l->uf);
205  layer_resize(l->wo, l->uf);
206  l->n_outputs = l->uf->n_outputs;
207  l->out_w = l->n_outputs;
208  l->out_c = 1;
209  l->out_h = 1;
210  set_layer_n_weights(l);
211  set_layer_n_biases(l);
212  set_layer_n_active(l);
213  free_layer_arrays(l);
214  malloc_layer_arrays(l);
215  return true;
216  }
217  return false;
218 }

268 {
269  l->options = layer_args_opt(args);
270  l->function = args->function;
271  l->recurrent_function = args->recurrent_function;
272  l->n_inputs = args->n_inputs;
273  l->n_outputs = args->n_init;
274  l->max_outputs = args->n_max;
275  l->out_w = l->n_outputs;

277  l->out_h = 1;
278  l->eta_max = args->eta;
279  l->momentum = args->momentum;
280  l->max_neuron_grow = args->max_neuron_grow;
281  l->decay = args->decay;
282  struct ArgsLayer *cargs = layer_args_copy(args);
283  cargs->type = CONNECTED; // lstm is composed of 8 connected layers
284  cargs->function = LINEAR;
285  l->uf = layer_init(cargs); // input layers
286  l->ui = layer_init(cargs);
287  l->ug = layer_init(cargs);
288  l->uo = layer_init(cargs);
289  cargs->n_inputs = cargs->n_init;
290  l->wf = layer_init(cargs); // self layers
291  l->wi = layer_init(cargs);
292  l->wg = layer_init(cargs);
293  l->wo = layer_init(cargs);
294  free(cargs);
295  set_layer_n_biases(l);
296  set_layer_n_weights(l);
297  set_layer_n_active(l);
298  set_eta(l);
299  malloc_layer_arrays(l);
300  l->mu = malloc(sizeof(double) * N_MU);
301  sam_init(l->mu, N_MU, MU_TYPE);
302 }

401 void
402 neural_layer_lstm_forward(const struct Layer *l, const struct Net *net,
403  const double *input)
404 {
405  layer_forward(l->uf, net, input);
406  layer_forward(l->ui, net, input);
407  layer_forward(l->ug, net, input);
408  layer_forward(l->uo, net, input);
409  layer_forward(l->wf, net, l->h);
410  layer_forward(l->wi, net, l->h);
411  layer_forward(l->wg, net, l->h);
412  layer_forward(l->wo, net, l->h);
413  memcpy(l->f, l->wf->output, sizeof(double) * l->n_outputs);
414  blas_axpy(l->n_outputs, 1, l->uf->output, 1, l->f, 1);
415  memcpy(l->i, l->wi->output, sizeof(double) * l->n_outputs);
416  blas_axpy(l->n_outputs, 1, l->ui->output, 1, l->i, 1);
417  memcpy(l->g, l->wg->output, sizeof(double) * l->n_outputs);
418  blas_axpy(l->n_outputs, 1, l->ug->output, 1, l->g, 1);
419  memcpy(l->o, l->wo->output, sizeof(double) * l->n_outputs);
420  blas_axpy(l->n_outputs, 1, l->uo->output, 1, l->o, 1);
421  neural_activate_array(l->f, l->f, l->n_outputs, l->recurrent_function);
422  neural_activate_array(l->i, l->i, l->n_outputs, l->recurrent_function);
423  neural_activate_array(l->g, l->g, l->n_outputs, l->function);
424  neural_activate_array(l->o, l->o, l->n_outputs, l->recurrent_function);
425  memcpy(l->temp, l->i, sizeof(double) * l->n_outputs);
426  blas_mul(l->n_outputs, l->g, 1, l->temp, 1);
427  blas_mul(l->n_outputs, l->f, 1, l->c, 1);
428  blas_axpy(l->n_outputs, 1, l->temp, 1, l->c, 1);
429  memcpy(l->h, l->c, sizeof(double) * l->n_outputs);
430  neural_activate_array(l->h, l->h, l->n_outputs, l->function);
431  blas_mul(l->n_outputs, l->o, 1, l->h, 1);
432  memcpy(l->cell, l->c, sizeof(double) * l->n_outputs);
433  memcpy(l->output, l->h, sizeof(double) * l->n_outputs);
434 }

443 void
444 neural_layer_lstm_backward(const struct Layer *l, const struct Net *net,
445  const double *input, double *delta)
446 {
447  reset_layer_deltas(l);
448  memcpy(l->temp3, l->delta, sizeof(double) * l->n_outputs);
449  memcpy(l->temp, l->c, sizeof(double) * l->n_outputs);
450  neural_activate_array(l->temp, l->temp, l->n_outputs, l->function);
451  memcpy(l->temp2, l->temp3, sizeof(double) * l->n_outputs);
452  blas_mul(l->n_outputs, l->o, 1, l->temp2, 1);
453  neural_gradient_array(l->temp, l->temp2, l->n_outputs, l->function);
454  blas_axpy(l->n_outputs, 1, l->dc, 1, l->temp2, 1);
455  memcpy(l->temp, l->c, sizeof(double) * l->n_outputs);
456  neural_activate_array(l->temp, l->temp, l->n_outputs, l->function);
457  blas_mul(l->n_outputs, l->temp3, 1, l->temp, 1);
458  neural_gradient_array(l->o, l->temp, l->n_outputs, l->recurrent_function);
459  memcpy(l->wo->delta, l->temp, sizeof(double) * l->n_outputs);
460  layer_backward(l->wo, net, l->prev_state, 0);
461  memcpy(l->uo->delta, l->temp, sizeof(double) * l->n_outputs);
462  layer_backward(l->uo, net, input, delta);
463  memcpy(l->temp, l->temp2, sizeof(double) * l->n_outputs);
464  blas_mul(l->n_outputs, l->i, 1, l->temp, 1);
465  neural_gradient_array(l->g, l->temp, l->n_outputs, l->function);
466  memcpy(l->wg->delta, l->temp, sizeof(double) * l->n_outputs);
467  layer_backward(l->wg, net, l->prev_state, 0);
468  memcpy(l->ug->delta, l->temp, sizeof(double) * l->n_outputs);
469  layer_backward(l->ug, net, input, delta);
470  memcpy(l->temp, l->temp2, sizeof(double) * l->n_outputs);
471  blas_mul(l->n_outputs, l->g, 1, l->temp, 1);
472  neural_gradient_array(l->i, l->temp, l->n_outputs, l->recurrent_function);
473  memcpy(l->wi->delta, l->temp, sizeof(double) * l->n_outputs);
474  layer_backward(l->wi, net, l->prev_state, 0);
475  memcpy(l->ui->delta, l->temp, sizeof(double) * l->n_outputs);
476  layer_backward(l->ui, net, input, delta);
477  memcpy(l->temp, l->temp2, sizeof(double) * l->n_outputs);
478  blas_mul(l->n_outputs, l->prev_cell, 1, l->temp, 1);
479  neural_gradient_array(l->f, l->temp, l->n_outputs, l->recurrent_function);
480  memcpy(l->wf->delta, l->temp, sizeof(double) * l->n_outputs);
481  layer_backward(l->wf, net, l->prev_state, 0);
482  memcpy(l->uf->delta, l->temp, sizeof(double) * l->n_outputs);
483  layer_backward(l->uf, net, input, delta);
484  memcpy(l->temp, l->temp2, sizeof(double) * l->n_outputs);
485  blas_mul(l->n_outputs, l->f, 1, l->temp, 1);
486  memcpy(l->dc, l->temp, sizeof(double) * l->n_outputs);
487 }

543 bool
544 neural_layer_lstm_mutate(struct Layer *l)
545 {
546  sam_adapt(l->mu, N_MU, MU_TYPE);
547  bool mod = false;
548  if ((l->options & LAYER_EVOLVE_ETA) && mutate_eta(l)) {
549  mod = true;
550  }
551  if ((l->options & LAYER_EVOLVE_NEURONS) && mutate_neurons(l)) {
552  mod = true;
553  }
554  if ((l->options & LAYER_EVOLVE_CONNECT) && mutate_connectivity(l)) {
555  mod = true;
556  }
557  if ((l->options & LAYER_EVOLVE_WEIGHTS) && mutate_weights(l)) {
558  mod = true;
559  }
560  if ((l->options & LAYER_EVOLVE_FUNCTIONS) &&
561  layer_mutate_functions(l, l->mu[5])) {
562  mod = true;
563  }
564  return mod;
565 }
