91 static void
92 action_param_defaults_neural(struct XCSF *xcsf)
93 {
94  // hidden layer
95  struct ArgsLayer *la = malloc(sizeof(struct ArgsLayer));
96  layer_args_init(la);
97  la->type = CONNECTED;
98  la->n_inputs = xcsf->x_dim;
99  la->n_init = 1;
100  la->n_max = 100;
101  la->max_neuron_grow = 1;
102  la->function = LOGISTIC;
103  la->evolve_weights = true;
104  la->evolve_neurons = true;
105  la->evolve_connect = true;
106  xcsf->act->largs = la;
107  // softmax output layer
108  la->next = layer_args_copy(la);
109  la->next->function = LINEAR;
110  la->next->n_inputs = la->n_init;
111  la->next->n_init = xcsf->n_actions;
112  la->next->n_max = xcsf->n_actions;
113  la->next->evolve_neurons = false;
114  la->next->next = layer_args_copy(la->next);
115  la->next->next->n_inputs = la->next->n_init;
116  la->next->next->type = SOFTMAX;
117  la->next->next->scale = 1;
118 }
