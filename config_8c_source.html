562 static void
563 config_newnvpair(struct XCSF *xcsf, const char *param)
564 {
565  // get length of parameter name
566  size_t namelen = 0;
567  bool err = true;
568  for (namelen = 0; namelen < strnlen(param, MAXLEN); ++namelen) {
569  if (param[namelen] == '=') {
570  err = false;
571  break;
572  }
573  }
574  if (err) {
575  return; // no '=' found
576  }
577  // get parameter name
578  char *name = malloc(namelen + 1);
579  for (size_t i = 0; i < namelen; ++i) {
580  name[i] = param[i];
581  }
582  name[namelen] = '\0';
583  // get parameter value
584  const size_t valuelen = strnlen(param, MAXLEN) - namelen;
585  char *value = malloc(valuelen + 1);
586  for (size_t i = 0; i < valuelen; ++i) {
587  value[i] = param[namelen + 1 + i];
588  }
589  value[valuelen] = '\0';
590  config_add_param(xcsf, name, value);
591  free(name);
592  free(value);
593 }

600 static void
601 config_process(struct XCSF *xcsf, const char *configline)
602 {
603  if (strnlen(configline, MAXLEN) == 0) { // ignore empty lines
604  return;
605  }
606  if (configline[0] == '#') { // lines starting with # are comments
607  return;
608  }
609  char *ptr = strchr(configline, '#'); // remove anything after #
610  if (ptr != NULL) {
611  *ptr = '\0';
612  }
613  config_newnvpair(xcsf, configline);
614 }

621 void
622 config_read(struct XCSF *xcsf, const char *filename)
623 {
624  FILE *f = fopen(filename, "rt");
625  if (f == NULL) {
626  printf("Warning: could not open %s. %s.\n", filename, strerror(errno));
627  return;
628  }
629  // clear existing layer parameters
630  layer_args_free(&xcsf->act->largs);
631  layer_args_free(&xcsf->cond->largs);
632  layer_args_free(&xcsf->pred->largs);
633  current_layer = NULL;
634  // load new parameters from config
635  char buff[MAXLEN];
636  while (!feof(f)) {
637  if (fgets(buff, MAXLEN - 2, f) == NULL) {
638  break;
639  }
640  config_trim(buff);
641  config_process(xcsf, buff);
642  }
643  fclose(f);
644  layer_args_validate(xcsf->act->largs);
645  layer_args_validate(xcsf->cond->largs);
646  layer_args_validate(xcsf->pred->largs);
647  tree_args_init_constants(xcsf->cond->targs);
648 }
