560 static void
561 config_newnvpair(struct XCSF *xcsf, const char *param)
562 {
563  // get length of parameter name
564  size_t namelen = 0;
565  bool err = true;
566  for (namelen = 0; namelen < strnlen(param, MAXLEN); ++namelen) {
567  if (param[namelen] == '=') {
568  err = false;
569  break;
570  }
571  }
572  if (err) {
573  return; // no '=' found
574  }
575  // get parameter name
576  char *name = malloc(namelen + 1);
577  for (size_t i = 0; i < namelen; ++i) {
578  name[i] = param[i];
579  }
580  name[namelen] = '\0';
581  // get parameter value
582  const size_t valuelen = strnlen(param, MAXLEN) - namelen;
583  char *value = malloc(valuelen + 1);
584  for (size_t i = 0; i < valuelen; ++i) {
585  value[i] = param[namelen + 1 + i];
586  }
587  value[valuelen] = '\0';
588  config_add_param(xcsf, name, value);
589  free(name);
590  free(value);
591 }

598 static void
599 config_process(struct XCSF *xcsf, const char *configline)
600 {
601  if (strnlen(configline, MAXLEN) == 0) { // ignore empty lines
602  return;
603  }
604  if (configline[0] == '#') { // lines starting with # are comments
605  return;
606  }
607  char *ptr = strchr(configline, '#'); // remove anything after #
608  if (ptr != NULL) {
609  *ptr = '\0';
610  }
611  config_newnvpair(xcsf, configline);
612 }

619 void
620 config_read(struct XCSF *xcsf, const char *filename)
621 {
622  FILE *f = fopen(filename, "rt");
623  if (f == NULL) {
624  printf("Warning: could not open %s.\n", filename);
625  perror("");
626  return;
627  }
628  // clear existing layer parameters
629  layer_args_free(&xcsf->act->largs);
630  layer_args_free(&xcsf->cond->largs);
631  layer_args_free(&xcsf->pred->largs);
632  current_layer = NULL;
633  // load new parameters from config
634  char buff[MAXLEN];
635  while (!feof(f)) {
636  if (fgets(buff, MAXLEN - 2, f) == NULL) {
637  break;
638  }
639  config_trim(buff);
640  config_process(xcsf, buff);
641  }
642  fclose(f);
643  layer_args_validate(xcsf->act->largs);
644  layer_args_validate(xcsf->cond->largs);
645  layer_args_validate(xcsf->pred->largs);
646  tree_args_init_constants(xcsf->cond->targs);
647 }
