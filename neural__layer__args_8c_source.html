184 static void
185 layer_args_json_export_activation(cJSON *json, const struct ArgsLayer *args)
186 {
187  switch (args->type) {
188  case AVGPOOL:
189  case MAXPOOL:
190  case UPSAMPLE:
191  case DROPOUT:
192  case NOISE:
193  case SOFTMAX:
194  return;
195  default:
196  break;
197  }
198  cJSON_AddStringToObject(json, "activation",
199  neural_activation_string(args->function));
200  if (args->type == LSTM) {
201  cJSON_AddStringToObject(
202  json, "recurrent_activation",
203  neural_activation_string(args->recurrent_function));
204  }
205 }

215 {

218  cJSON_AddNumberToObject(json, "probability", args->probability);

221  if (args->type == NOISE || args->type == SOFTMAX) {
222  cJSON_AddNumberToObject(json, "scale", args->scale);
223  cont = true;

225  if (args->type == MAXPOOL) {
226  cont = true;
227  }

236 const char *
237 layer_args_json_export(struct ArgsLayer *args)
238 {
239  struct Net net; // create a temporary network to parse inputs
240  neural_init(&net);
241  neural_create(&net, args);
242  neural_free(&net);
243  cJSON *json = cJSON_CreateObject();
244  int cnt = 0;
245  for (const struct ArgsLayer *a = args; a != NULL; a = a->next) {
246  char name[256];
247  snprintf(name, 256, "layer_%d", cnt);
248  ++cnt;
249  cJSON *l = cJSON_CreateObject();
250  cJSON_AddItemToObject(json, name, l);
251  cJSON_AddStringToObject(l, "type", layer_type_as_string(a->type));
252  layer_args_json_export_activation(l, a);
253  layer_args_json_export_inputs(l, a);
254  if (layer_args_json_export_scale(l, a)) {
255  continue;
256  }
257  if (a->n_init > 0) {
258  cJSON_AddNumberToObject(l, "n_init", a->n_init);
259  }
260  layer_args_json_export_evo(l, a);
261  layer_args_json_export_sgd(l, a);
262  }
263  const char *string = cJSON_Print(json);
264  cJSON_Delete(json);
265  return string;
266 }

286 static void
287 layer_args_validate_inputs(struct ArgsLayer *arg)
288 {
289  if (arg->type == DROPOUT || arg->type == NOISE) {
290  if (arg->n_inputs < 1) {
291  arg->n_inputs = arg->channels * arg->height * arg->width;
292  } else if (arg->channels < 1 || arg->height < 1 || arg->width < 1) {
293  arg->channels = 1;
294  arg->height = 1;
295  arg->width = arg->n_inputs;
296  }
297  }
298  if (layer_receives_images(arg->type)) {
299  if (arg->channels < 1) {
300  printf("Error: input channels < 1\n");
301  exit(EXIT_FAILURE);
302  }
303  if (arg->height < 1) {
304  printf("Error: input height < 1\n");
305  exit(EXIT_FAILURE);
306  }
307  if (arg->width < 1) {
308  printf("Error: input width < 1\n");
309  exit(EXIT_FAILURE);
310  }
311  } else if (arg->n_inputs < 1) {
312  printf("Error: number of inputs < 1\n");
313  exit(EXIT_FAILURE);
314  }
315 }

321 void
322 layer_args_validate(struct ArgsLayer *args)
323 {
324  struct ArgsLayer *arg = args;
325  if (arg == NULL) {
326  printf("Error: empty layer argument list\n");
327  exit(EXIT_FAILURE);
328  }
329  layer_args_validate_inputs(arg);
330  do {
331  if (arg->evolve_neurons && arg->max_neuron_grow < 1) {
332  printf("Error: evolving neurons but max_neuron_grow < 1\n");
333  exit(EXIT_FAILURE);
334  }
335  if (arg->n_max < arg->n_init) {
336  arg->n_max = arg->n_init;
337  }
338  arg = arg->next;
339  } while (arg != NULL);
340 }
