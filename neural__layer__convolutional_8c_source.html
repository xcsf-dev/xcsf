259 void
260 neural_layer_convolutional_forward(const struct Layer *l, const struct Net *net,
261  const double *input)
262 {
263  (void) net;
264  const int m = l->n_filters;
265  const int k = l->size * l->size * l->channels;
266  const int n = l->out_w * l->out_h;
267  const double *a = l->weights;
268  double *b = l->temp;
269  double *c = l->state;
270  memset(l->state, 0, sizeof(double) * l->n_outputs);
271  if (l->size == 1) {
272  blas_gemm(0, 0, m, n, k, 1, a, k, input, n, 1, c, n);
273  } else {
274  im2col(input, l->channels, l->height, l->width, l->size, l->stride,
275  l->pad, b);
276  blas_gemm(0, 0, m, n, k, 1, a, k, b, n, 1, c, n);
277  }
278  for (int i = 0; i < l->n_biases; ++i) {
279  for (int j = 0; j < n; ++j) {
280  l->state[i * n + j] += l->biases[i];
281  }
282  }
283  neural_activate_array(l->state, l->output, l->n_outputs, l->function);
284 }

293 void
294 neural_layer_convolutional_backward(const struct Layer *l,
295  const struct Net *net, const double *input,
296  double *delta)
297 {
298  (void) net;
299  const int m = l->n_filters;
300  const int n = l->size * l->size * l->channels;
301  const int k = l->out_w * l->out_h;
302  if (l->options & LAYER_SGD_WEIGHTS) {
303  neural_gradient_array(l->state, l->delta, l->n_outputs, l->function);
304  for (int i = 0; i < l->n_biases; ++i) {
305  l->bias_updates[i] += blas_sum(l->delta + k * i, k);
306  }
307  const double *a = l->delta;
308  double *b = l->temp;
309  double *c = l->weight_updates;
310  if (l->size == 1) {
311  blas_gemm(0, 1, m, n, k, 1, a, k, input, k, 1, c, n);
312  } else {
313  im2col(input, l->channels, l->height, l->width, l->size, l->stride,
314  l->pad, b);
315  blas_gemm(0, 1, m, n, k, 1, a, k, b, k, 1, c, n);
316  }
317  }
318  if (delta) {
319  const double *a = l->weights;
320  const double *b = l->delta;
321  double *c = l->temp;
322  if (l->size == 1) {
323  c = delta;
324  }
325  blas_gemm(1, 0, n, k, m, 1, a, n, b, k, 0, c, k);
326  if (l->size != 1) {
327  col2im(l->temp, l->channels, l->height, l->width, l->size,
328  l->stride, l->pad, delta);
329  }
330  }
331 }

337 void
338 neural_layer_convolutional_update(const struct Layer *l)
339 {
340  if (l->options & LAYER_SGD_WEIGHTS && l->eta > 0) {
341  blas_axpy(l->n_biases, l->eta, l->bias_updates, 1, l->biases, 1);
342  blas_scal(l->n_biases, l->momentum, l->bias_updates, 1);
343  if (l->decay > 0) {
344  blas_axpy(l->n_weights, -(l->decay), l->weights, 1,
345  l->weight_updates, 1);
346  }
347  blas_axpy(l->n_weights, l->eta, l->weight_updates, 1, l->weights, 1);
348  blas_scal(l->n_weights, l->momentum, l->weight_updates, 1);
349  layer_weight_clamp(l);
350  }
351 }

391 static int
392 neural_layer_convolutional_mutate_filter(const struct Layer *l, const double mu)
393 {
394  int n = 0;
395  if (rand_uniform(0, 0.1) < mu) { // 10x higher probability
396  while (n == 0) {
397  const double m = clamp(rand_normal(0, 0.5), -1, 1);
398  n = (int) round(m * l->max_neuron_grow);
399  }
400  if (l->n_filters + n < 1) {
401  n = -(l->n_filters - 1);
402  } else if (l->n_filters + n > l->max_outputs) {
403  n = l->max_outputs - l->n_filters;
404  }
405  }
406  return n;
407 }

415 static void
416 neural_layer_convolutional_add_filters(struct Layer *l, const int N)
417 {
418  const int old_n_biases = l->n_biases;
419  const int old_n_weights = l->n_weights;
420  const int old_n_outputs = l->n_outputs;
421  l->n_filters += N;
422  l->n_biases = l->n_filters;
423  l->out_c = l->n_filters;
424  l->n_weights = l->channels * l->n_filters * l->size * l->size;
425  l->n_outputs = l->out_h * l->out_w * l->out_c;
426  realloc_layer_arrays(l);
427  for (int i = old_n_weights; i < l->n_weights; ++i) {
428  l->weights[i] = rand_normal(0, WEIGHT_SD);
429  l->weight_active[i] = true;
430  l->weight_updates[i] = 0;
431  }
432  for (int i = old_n_biases; i < l->n_biases; ++i) {
433  l->biases[i] = 0;
434  l->bias_updates[i] = 0;
435  }
436  for (int i = old_n_outputs; i < l->n_outputs; ++i) {
437  l->output[i] = 0;
438  l->state[i] = 0;
439  l->delta[i] = 0;
440  }
441  layer_calc_n_active(l);
442 }

449 bool
450 neural_layer_convolutional_mutate(struct Layer *l)
451 {
452  sam_adapt(l->mu, N_MU, MU_TYPE);
453  bool mod = false;
454  if ((l->options & LAYER_EVOLVE_ETA) && layer_mutate_eta(l, l->mu[0])) {
455  mod = true;
456  }
457  if (l->options & LAYER_EVOLVE_NEURONS) {
458  const int n = neural_layer_convolutional_mutate_filter(l, l->mu[1]);
459  if (n != 0) {
460  neural_layer_convolutional_add_filters(l, n);
461  mod = true;
462  }
463  }
464  if ((l->options & LAYER_EVOLVE_CONNECT) &&
465  layer_mutate_connectivity(l, l->mu[2], l->mu[3])) {
466  mod = true;
467  }
468  if ((l->options & LAYER_EVOLVE_WEIGHTS) &&
469  layer_mutate_weights(l, l->mu[4])) {
470  mod = true;
471  }
472  if ((l->options & LAYER_EVOLVE_FUNCTIONS) &&
473  layer_mutate_functions(l, l->mu[5])) {
474  mod = true;
475  }
476  return mod;
477 }
