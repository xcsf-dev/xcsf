38 double
39 clset_mean_cond_layers(const struct XCSF *xcsf, const struct Set *set)
40 {
41  int sum = 0;
42  int cnt = 0;
43  if (xcsf->cond->type == COND_TYPE_NEURAL ||
44  xcsf->cond->type == RULE_TYPE_NEURAL) {
45  const struct Clist *iter = set->list;
46  while (iter != NULL) {
47  sum += cond_neural_layers(xcsf, iter->cl);
48  ++cnt;
49  iter = iter->next;
50  }
51  }
52  if (cnt != 0) {
53  return (double) sum / cnt;
54  }
55  return 0;
56 }
