110 {
111  if (f >= 1) {
112  for (int i = 0; i < bits; ++i) {
113  binary[i] = '1';
114  }
115  } else if (f <= 0) {
116  for (int i = 0; i < bits; ++i) {
117  binary[i] = '0';
118  }
119  } else {
120  int a = (int) (f * pow(2, bits));
121  for (int i = 0; i < bits; ++i) {
122  binary[bits - 1 - i] = (a % 2) + '0';

124  }
