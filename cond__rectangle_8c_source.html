45 static double
46 cond_rectangle_dist(const struct XCSF *xcsf, const struct Cl *c,
47  const double *x)
48 {
49  const struct CondRectangle *cond = c->cond;
50  double dist = 0;
51  for (int i = 0; i < xcsf->x_dim; ++i) {
52  const double d = fabs((x[i] - cond->center[i]) / cond->spread[i]);
53  if (d > dist) {
54  dist = d;
55  }
56  }
57  return dist;
58 }

178 bool
179 cond_rectangle_crossover(const struct XCSF *xcsf, const struct Cl *c1,
180  const struct Cl *c2)
181 {
182  const struct CondRectangle *cond1 = c1->cond;
183  const struct CondRectangle *cond2 = c2->cond;
184  bool changed = false;
185  if (rand_uniform(0, 1) < xcsf->ea->p_crossover) {
186  for (int i = 0; i < xcsf->x_dim; ++i) {
187  if (rand_uniform(0, 1) < 0.5) {
188  const double tmp = cond1->center[i];
189  cond1->center[i] = cond2->center[i];
190  cond2->center[i] = tmp;
191  changed = true;
192  }
193  if (rand_uniform(0, 1) < 0.5) {
194  const double tmp = cond1->spread[i];
195  cond1->spread[i] = cond2->spread[i];
196  cond2->spread[i] = tmp;
197  changed = true;
198  }
199  }
200  }
201  return changed;
202 }

210 bool
211 cond_rectangle_mutate(const struct XCSF *xcsf, const struct Cl *c)
212 {
213  bool changed = false;
214  const struct CondRectangle *cond = c->cond;
215  double *center = cond->center;
216  double *spread = cond->spread;
217  sam_adapt(cond->mu, N_MU, MU_TYPE);
218  for (int i = 0; i < xcsf->x_dim; ++i) {
219  double orig = center[i];
220  center[i] += rand_normal(0, cond->mu[0]);
221  center[i] = clamp(center[i], xcsf->cond->min, xcsf->cond->max);
222  if (orig != center[i]) {
223  changed = true;
224  }
225  orig = spread[i];
226  spread[i] += rand_normal(0, cond->mu[0]);
227  spread[i] = fmax(DBL_EPSILON, spread[i]);
228  if (orig != spread[i]) {
229  changed = true;
230  }
231  }
232  return changed;
233 }

242 bool
243 cond_rectangle_general(const struct XCSF *xcsf, const struct Cl *c1,
244  const struct Cl *c2)
245 {
246  const struct CondRectangle *cond1 = c1->cond;
247  const struct CondRectangle *cond2 = c2->cond;
248  for (int i = 0; i < xcsf->x_dim; ++i) {
249  const double l1 = cond1->center[i] - cond1->spread[i];
250  const double l2 = cond2->center[i] - cond2->spread[i];
251  const double u1 = cond1->center[i] + cond1->spread[i];
252  const double u2 = cond2->center[i] + cond2->spread[i];
253  if (l1 > l2 || u1 < u2) {
254  return false;
255  }
256  }
257  return true;
258 }
