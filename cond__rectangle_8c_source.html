45 static double
46 cond_rectangle_dist(const struct XCSF *xcsf, const struct Cl *c,
47  const double *x)
48 {
49  const struct CondRectangle *cond = c->cond;
50  double dist = 0;
51  for (int i = 0; i < xcsf->x_dim; ++i) {
52  const double d = fabs((x[i] - cond->b1[i]) / cond->b2[i]);
53  if (d > dist) {
54  dist = d;
55  }
56  }
57  return dist;
58 }

65 void
66 cond_rectangle_init(const struct XCSF *xcsf, struct Cl *c)
67 {
68  struct CondRectangle *new = malloc(sizeof(struct CondRectangle));
69  new->b1 = malloc(sizeof(double) * xcsf->x_dim);
70  new->b2 = malloc(sizeof(double) * xcsf->x_dim);
71  const double spread_max = fabs(xcsf->cond->max - xcsf->cond->min);
72  for (int i = 0; i < xcsf->x_dim; ++i) {
73  new->b1[i] = rand_uniform(xcsf->cond->min, xcsf->cond->max);
74  new->b2[i] = rand_uniform(xcsf->cond->min, xcsf->cond->max);
75  }
76  if (xcsf->cond->type == COND_TYPE_HYPERRECTANGLE_CSR) {
77  // csr: b1 = center, b2 = spread
78  for (int i = 0; i < xcsf->x_dim; ++i) {
79  new->b2[i] = rand_uniform(xcsf->cond->spread_min, spread_max);
80  }
81  }
82  new->mu = malloc(sizeof(double) * N_MU);
83  sam_init(new->mu, N_MU, MU_TYPE);
84  c->cond = new;
85 }

130 void
131 cond_rectangle_cover(const struct XCSF *xcsf, const struct Cl *c,
132  const double *x)
133 {
134  const struct CondRectangle *cond = c->cond;
135  const double spread_max = fabs(xcsf->cond->max - xcsf->cond->min);
136  if (xcsf->cond->type == COND_TYPE_HYPERRECTANGLE_CSR) {
137  for (int i = 0; i < xcsf->x_dim; ++i) {
138  cond->b1[i] = x[i];
139  cond->b2[i] = rand_uniform(xcsf->cond->spread_min, spread_max);
140  }
141  } else {
142  for (int i = 0; i < xcsf->x_dim; ++i) {
143  const double r1 = rand_uniform(xcsf->cond->spread_min, spread_max);
144  const double r2 = rand_uniform(xcsf->cond->spread_min, spread_max);
145  cond->b1[i] = x[i] - (r1 * 0.5);
146  cond->b2[i] = x[i] + (r2 * 0.5);
147  }
148  }
149 }

179 bool
180 cond_rectangle_match(const struct XCSF *xcsf, const struct Cl *c,
181  const double *x)
182 {
183  const struct CondRectangle *cond = c->cond;
184  if (xcsf->cond->type == COND_TYPE_HYPERRECTANGLE_CSR) {
185  return (cond_rectangle_dist(xcsf, c, x) < 1);
186  } else { // ubr
187  for (int i = 0; i < xcsf->x_dim; ++i) {
188  const double lb = fmin(cond->b1[i], cond->b2[i]);
189  const double ub = fmax(cond->b1[i], cond->b2[i]);
190  if (x[i] < lb || x[i] > ub) {
191  return false;
192  }
193  }
194  }
195  return true;
196 }

205 bool
206 cond_rectangle_crossover(const struct XCSF *xcsf, const struct Cl *c1,
207  const struct Cl *c2)
208 {
209  const struct CondRectangle *cond1 = c1->cond;
210  const struct CondRectangle *cond2 = c2->cond;
211  bool changed = false;
212  if (rand_uniform(0, 1) < xcsf->ea->p_crossover) {
213  for (int i = 0; i < xcsf->x_dim; ++i) {
214  if (rand_uniform(0, 1) < 0.5) {
215  const double tmp = cond1->b1[i];
216  cond1->b1[i] = cond2->b1[i];
217  cond2->b1[i] = tmp;
218  changed = true;
219  }
220  if (rand_uniform(0, 1) < 0.5) {
221  const double tmp = cond1->b2[i];
222  cond1->b2[i] = cond2->b2[i];
223  cond2->b2[i] = tmp;
224  changed = true;
225  }
226  }
227  }
228  return changed;
229 }

237 bool
238 cond_rectangle_mutate(const struct XCSF *xcsf, const struct Cl *c)
239 {
240  bool changed = false;
241  const struct CondRectangle *cond = c->cond;
242  double *b1 = cond->b1;
243  double *b2 = cond->b2;
244  sam_adapt(cond->mu, N_MU, MU_TYPE);
245  for (int i = 0; i < xcsf->x_dim; ++i) {
246  double orig = b1[i];
247  b1[i] += rand_normal(0, cond->mu[0]);
248  b1[i] = clamp(b1[i], xcsf->cond->min, xcsf->cond->max);
249  if (orig != b1[i]) {
250  changed = true;
251  }
252  orig = b2[i];
253  b2[i] += rand_normal(0, cond->mu[0]);
254  if (xcsf->cond->type == COND_TYPE_HYPERRECTANGLE_CSR) {
255  b2[i] = fmax(DBL_EPSILON, b2[i]);
256  } else {
257  b2[i] = clamp(b2[i], xcsf->cond->min, xcsf->cond->max);
258  }
259  if (orig != b2[i]) {
260  changed = true;
261  }
262  }
263  return changed;
264 }

273 bool
274 cond_rectangle_general(const struct XCSF *xcsf, const struct Cl *c1,
275  const struct Cl *c2)
276 {
277  const struct CondRectangle *cond1 = c1->cond;
278  const struct CondRectangle *cond2 = c2->cond;
279  if (xcsf->cond->type == COND_TYPE_HYPERRECTANGLE_CSR) {
280  for (int i = 0; i < xcsf->x_dim; ++i) {
281  const double l1 = cond1->b1[i] - cond1->b2[i];
282  const double l2 = cond2->b1[i] - cond2->b2[i];
283  const double u1 = cond1->b1[i] + cond1->b2[i];
284  const double u2 = cond2->b1[i] + cond2->b2[i];
285  if (l1 > l2 || u1 < u2) {
286  return false;
287  }
288  }
289  } else {
290  for (int i = 0; i < xcsf->x_dim; ++i) {
291  const double l1 = fmin(cond1->b1[i], cond1->b2[i]);
292  const double l2 = fmin(cond2->b1[i], cond2->b2[i]);
293  const double u1 = fmax(cond1->b1[i], cond1->b2[i]);
294  const double u2 = fmax(cond2->b1[i], cond2->b2[i]);
295  if (l1 > l2 || u1 < u2) {
296  return false;
297  }
298  }
299  }
300  return true;
301 }

418  for (int i = 0; i < xcsf->x_dim; ++i) {
419  const cJSON *item_i = cJSON_GetArrayItem(item, i);
420  cond->b1[i] = item_i->valuedouble;
