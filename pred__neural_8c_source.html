303 void
304 pred_neural_expand(const struct XCSF *xcsf, const struct Cl *c)
305 {
306  struct PredNeural *pred = c->pred;
307  struct Net *net = &pred->net;
308  const struct Layer *h = NULL;
309  int n_inputs = 0;
310  if (net->n_layers > 1) { // select top hidden layer
311  h = net->head->next->layer;
312  n_inputs = h->n_outputs;
313  } else { // if only one layer, must use output layer
314  h = net->head->layer;
315  n_inputs = h->n_inputs;
316  }
317  const struct ArgsLayer *largs = xcsf->pred->largs;
318  struct ArgsLayer new;
319  layer_args_init(&new);
320  new.type = CONNECTED;
321  new.function = largs->function;
322  new.n_inputs = n_inputs;
323  new.n_init = h->n_outputs;
324  new.n_max = h->max_outputs;
325  new.evolve_connect = largs->evolve_connect;
326  new.evolve_weights = largs->evolve_weights;
327  new.evolve_functions = largs->evolve_functions;
328  new.evolve_eta = largs->evolve_eta;
329  new.sgd_weights = largs->sgd_weights;
330  new.eta = largs->eta;
331  new.eta_min = largs->eta_min;
332  new.momentum = largs->momentum;
333  new.decay = largs->decay;
334  struct Layer *l = layer_init(&new);
335  const int pos = net->n_layers - 1;
336  neural_insert(net, l, pos);
337  neural_resize(net);
338 }

346 void
347 pred_neural_ae_to_classifier(const struct XCSF *xcsf, const struct Cl *c,
348  const int n_del)
349 {
350  struct PredNeural *pred = c->pred;
351  struct Net *net = &pred->net;
352  struct Layer *l = NULL;
353  // remove decoder layers
354  for (int i = 0; i < n_del && net->n_layers > 1; ++i) {
355  neural_pop(net);
356  }
357  // add new softmax output
358  const struct ArgsLayer *largs = xcsf->pred->largs;
359  struct ArgsLayer new;
360  layer_args_init(&new);
361  new.type = CONNECTED;
362  new.function = LINEAR;
363  new.n_inputs = net->n_outputs;
364  new.n_init = xcsf->y_dim;
365  new.n_max = xcsf->y_dim;
366  new.evolve_connect = largs->evolve_connect;
367  new.evolve_weights = largs->evolve_weights;
368  new.evolve_eta = largs->evolve_eta;
369  new.sgd_weights = largs->sgd_weights;
370  new.eta = largs->eta;
371  new.eta_min = largs->eta_min;
372  new.momentum = largs->momentum;
373  new.decay = largs->decay;
374  l = layer_init(&new);
375  neural_push(net, l);
376  new.type = SOFTMAX;
377  new.n_inputs = xcsf->y_dim;
378  new.scale = 1;
379  l = layer_init(&new);
380  neural_push(net, l);
381 }
