304 void
305 pred_neural_expand(const struct XCSF *xcsf, const struct Cl *c)
306 {
307  struct PredNeural *pred = c->pred;
308  struct Net *net = &pred->net;
309  const struct Layer *h = NULL;
310  int n_inputs = 0;
311  if (net->n_layers > 1) { // select top hidden layer
312  h = net->head->next->layer;
313  n_inputs = h->n_outputs;
314  } else { // if only one layer, must use output layer
315  h = net->head->layer;
316  n_inputs = h->n_inputs;
317  }
318  const struct ArgsLayer *largs = xcsf->pred->largs;
319  struct ArgsLayer new;
320  layer_args_init(&new);
321  new.type = CONNECTED;
322  new.function = largs->function;
323  new.n_inputs = n_inputs;
324  new.n_init = h->n_outputs;
325  new.n_max = h->max_outputs;
326  new.evolve_connect = largs->evolve_connect;
327  new.evolve_weights = largs->evolve_weights;
328  new.evolve_functions = largs->evolve_functions;
329  new.evolve_eta = largs->evolve_eta;
330  new.sgd_weights = largs->sgd_weights;
331  new.eta = largs->eta;
332  new.eta_min = largs->eta_min;
333  new.momentum = largs->momentum;
334  new.decay = largs->decay;
335  struct Layer *l = layer_init(&new);
336  const int pos = net->n_layers - 1;
337  neural_insert(net, l, pos);
338  neural_resize(net);
339 }

348 pred_neural_ae_to_classifier(const struct XCSF *xcsf, const struct Cl *c,
349  const int n_del)
350 {
351  struct PredNeural *pred = c->pred;
352  struct Net *net = &pred->net;
353  struct Layer *l = NULL;
354  // remove decoder layers
355  for (int i = 0; i < n_del && net->n_layers > 1; ++i) {
356  neural_pop(net);
357  }
358  // add new softmax output
359  const struct ArgsLayer *largs = xcsf->pred->largs;
360  struct ArgsLayer new;
361  layer_args_init(&new);
362  new.type = CONNECTED;
363  new.function = LINEAR;
364  new.n_inputs = net->n_outputs;
365  new.n_init = xcsf->y_dim;
366  new.n_max = xcsf->y_dim;
367  new.evolve_connect = largs->evolve_connect;
368  new.evolve_weights = largs->evolve_weights;
369  new.evolve_eta = largs->evolve_eta;
370  new.sgd_weights = largs->sgd_weights;
371  new.eta = largs->eta;
372  new.eta_min = largs->eta_min;
373  new.momentum = largs->momentum;
374  new.decay = largs->decay;
375  l = layer_init(&new);
376  neural_push(net, l);
