336  void
337  load_input(struct Input *data, const py::array_t<double> X,
338  const py::array_t<double> Y)
339  {
340  const py::buffer_info buf_x = X.request();
341  const py::buffer_info buf_y = Y.request();
342  if (buf_x.ndim < 1 || buf_x.ndim > 2) {
343  std::string error = "load_input(): X must be 1 or 2-D array";
344  throw std::invalid_argument(error);
345  }
346  if (buf_y.ndim < 1 || buf_y.ndim > 2) {
347  std::string error = "load_input(): Y must be 1 or 2-D array";
348  throw std::invalid_argument(error);
349  }
350  if (buf_x.shape[0] != buf_y.shape[0]) {
351  std::string error = "load_input(): X and Y n_samples are not equal";
352  throw std::invalid_argument(error);
353  }
354  if (buf_x.ndim > 1 && buf_x.shape[1] != xcs.x_dim) {
355  std::ostringstream error;
356  error << "load_input():";
357  error << " received x_dim: (" << buf_x.shape[1] << ")";
358  error << " but expected (" << xcs.x_dim << ")" << std::endl;
359  error << "Perhaps reshape your data.";
360  throw std::invalid_argument(error.str());
361  }
362  if (buf_y.ndim > 1 && buf_y.shape[1] != xcs.y_dim) {
363  std::ostringstream error;
364  error << "load_input():";
365  error << " received y_dim: (" << buf_y.shape[1] << ")";
366  error << " but expected (" << xcs.y_dim << ")" << std::endl;
367  error << "Perhaps reshape your data.";
368  throw std::invalid_argument(error.str());
369  }
370  data->n_samples = buf_x.shape[0];
371  data->x_dim = xcs.x_dim;
372  data->y_dim = xcs.y_dim;
373  data->x = (double *) buf_x.ptr;
374  data->y = (double *) buf_y.ptr;
375  }

447  bool
448  callbacks_run(py::list callbacks)
449  {
450  bool terminate = false;
451  py::dict metrics = get_metrics();
452  for (py::handle item : callbacks) {
453  if (py::isinstance<Callback>(item)) {
454  Callback *cb = py::cast<Callback *>(item);
455  if (cb->run(&xcs, metrics)) {
456  terminate = true;
457  }
458  } else {
459  std::ostringstream err;
460  err << "unsupported callback" << std::endl;
461  throw std::invalid_argument(err.str());
462  }
463  }
464  return terminate;
465  }

498  XCS &
499  fit(const py::array_t<double> X_train, const py::array_t<double> y_train,
500  const bool shuffle, const bool warm_start, const bool verbose,
501  py::object callbacks, py::kwargs kwargs)
502  {
503  if (!warm_start) { // re-initialise XCSF as necessary
504  xcsf_free(&xcs);
505  xcsf_init(&xcs);
506  }
507  load_input(train_data, X_train, y_train);
508  load_validation_data(kwargs);
509  // get callbacks
510  py::list calls;
511  if (py::isinstance<py::list>(callbacks)) {
512  calls = callbacks.cast<py::list>();
513  }
514  // break up the learning into epochs to track metrics
515  const int n = ceil(xcs.MAX_TRIALS / (double) xcs.PERF_TRIALS);
516  const int n_trials = std::min(xcs.MAX_TRIALS, xcs.PERF_TRIALS);
517  for (int i = 0; i < n; ++i) {
518  const int start = i * n_trials;
519  const double train_error = xcs_supervised_fit(
520  &xcs, train_data, NULL, shuffle, start, n_trials);
521  double val_error = 0;
522  if (val_data != NULL) {
523  val_error = xcs_supervised_score(&xcs, val_data, xcs.cover);
524  }
525  update_metrics(train_error, val_error, n_trials);
526  if (verbose) {
527  print_status();
528  }
529  if (callbacks_run(calls)) {
530  break;
531  }
532  }
533  callbacks_finish(calls);
534  return *this;
535  }
