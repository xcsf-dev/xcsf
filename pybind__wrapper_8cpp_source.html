330  void
331  load_input(struct Input *data, const py::array_t<double> X,
332  const py::array_t<double> Y)
333  {
334  const py::buffer_info buf_x = X.request();
335  const py::buffer_info buf_y = Y.request();
336  if (buf_x.shape[0] != buf_y.shape[0]) {
337  std::string error =
338  "load_input(): X and Y n_samples are not equal.";
339  throw std::invalid_argument(error);
340  }
341  if (buf_x.shape[1] != xcs.x_dim) {
342  std::ostringstream error;
343  error << "load_input(): x_dim != " << xcs.x_dim << std::endl;
344  error << "2-D arrays are required. Perhaps reshape your data.";
345  throw std::invalid_argument(error.str());
346  }
347  if (buf_y.shape[1] != xcs.y_dim) {
348  std::ostringstream error;
349  error << "load_input(): y_dim != " << xcs.y_dim << std::endl;
350  error << "2-D arrays are required. Perhaps reshape your data.";
351  throw std::invalid_argument(error.str());
352  }
353  data->n_samples = buf_x.shape[0];
354  data->x_dim = buf_x.shape[1];
355  data->y_dim = buf_y.shape[1];
356  data->x = (double *) buf_x.ptr;
357  data->y = (double *) buf_y.ptr;
358  }
