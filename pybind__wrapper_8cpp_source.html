367  double
368  fit(const py::array_t<double> train_X, const py::array_t<double> train_Y,
369  const py::array_t<double> test_X, const py::array_t<double> test_Y,
370  const bool shuffle)
371  {
372  const py::buffer_info buf_train_x = train_X.request();
373  const py::buffer_info buf_train_y = train_Y.request();
374  const py::buffer_info buf_test_x = test_X.request();
375  const py::buffer_info buf_test_y = test_Y.request();
376  if (buf_train_x.shape[0] != buf_train_y.shape[0]) {
377  printf("error: training X and Y n_samples are not equal\n");
378  exit(EXIT_FAILURE);
379  }
380  if (buf_test_x.shape[0] != buf_test_y.shape[0]) {

382  exit(EXIT_FAILURE);
383  }

385  printf("error: number of train and test X cols are not equal\n");
386  exit(EXIT_FAILURE);
387  }
388  if (buf_train_y.shape[1] != buf_test_y.shape[1]) {
389  printf("error: number of train and test Y cols are not equal\n");
390  exit(EXIT_FAILURE);
391  }
392  // load training data
393  train_data->n_samples = buf_train_x.shape[0];
394  train_data->x_dim = buf_train_x.shape[1];
395  train_data->y_dim = buf_train_y.shape[1];
396  train_data->x = (double *) buf_train_x.ptr;
397  train_data->y = (double *) buf_train_y.ptr;
398  // load testing data
399  test_data->n_samples = buf_test_x.shape[0];
400  test_data->x_dim = buf_test_x.shape[1];
401  test_data->y_dim = buf_test_y.shape[1];
402  test_data->x = (double *) buf_test_x.ptr;
403  test_data->y = (double *) buf_test_y.ptr;
404  // first execution
405  if (xcs.time == 0) {
406  clset_pset_init(&xcs);

408  // execute
409  return xcs_supervised_fit(&xcs, train_data, test_data, shuffle);
