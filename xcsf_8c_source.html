109 size_t
110 xcsf_load(struct XCSF *xcsf, const char *filename)
111 {
112  if (xcsf->pset.size > 0) {
113  clset_kill(xcsf, &xcsf->pset);
114  clset_init(&xcsf->pset);
115  }
116  FILE *fp = fopen(filename, "rb");
117  if (fp == 0) {
118  printf("Error loading file: %s. %s.\n", filename, strerror(errno));
119  exit(EXIT_FAILURE);
120  }
121  size_t s = 0;
122  int major = 0;
123  int minor = 0;
124  int build = 0;
125  s += fread(&major, sizeof(int), 1, fp);
126  s += fread(&minor, sizeof(int), 1, fp);
127  s += fread(&build, sizeof(int), 1, fp);
128  if (major != VERSION_MAJOR || minor != VERSION_MINOR) {
129  printf("Error loading file: %s. Version mismatch. ", filename);
130  printf("This version: %d.%d\n", VERSION_MAJOR, VERSION_MINOR);
131  printf("Loaded version: %d.%d\n", major, minor);
132  fclose(fp);
133  exit(EXIT_FAILURE);
134  }
135  s += param_load(xcsf, fp);
136  s += clset_pset_load(xcsf, fp);
137  fclose(fp);
138  return s;
139 }

166 void
167 xcsf_ae_to_classifier(struct XCSF *xcsf, const int y_dim, const int n_del)
168 {
169  pa_free(xcsf);
170  param_set_y_dim(xcsf, y_dim);
171  param_set_loss_func(xcsf, LOSS_ONEHOT);
172  pa_init(xcsf);
173  const struct Clist *iter = xcsf->pset.list;
174  while (iter != NULL) {
175  free(iter->cl->prediction);
176  iter->cl->prediction = calloc(xcsf->y_dim, sizeof(double));
177  pred_neural_ae_to_classifier(xcsf, iter->cl, n_del);
178  iter->cl->fit = xcsf->INIT_FITNESS;
179  iter->cl->err = xcsf->INIT_ERROR;
180  iter->cl->exp = 0;
181  iter->cl->time = xcsf->time;
182  iter = iter->next;
183  }

192 {

200  iter = iter->next;
201  }
