108 size_t
109 xcsf_load(struct XCSF *xcsf, const char *filename)
110 {
111  if (xcsf->pset.size > 0) {
112  clset_kill(xcsf, &xcsf->pset);
113  clset_init(&xcsf->pset);
114  }
115  FILE *fp = fopen(filename, "rb");
116  if (fp == 0) {
117  printf("Error loading file: %s. %s.\n", filename, strerror(errno));
118  exit(EXIT_FAILURE);
119  }
120  size_t s = 0;
121  int major = 0;
122  int minor = 0;
123  int build = 0;
124  s += fread(&major, sizeof(int), 1, fp);
125  s += fread(&minor, sizeof(int), 1, fp);
126  s += fread(&build, sizeof(int), 1, fp);
127  if (major != VERSION_MAJOR || minor != VERSION_MINOR) {
128  printf("Error loading file: %s. Version mismatch. ", filename);
129  printf("This version: %d.%d\n", VERSION_MAJOR, VERSION_MINOR);
130  printf("Loaded version: %d.%d\n", major, minor);
131  fclose(fp);
132  exit(EXIT_FAILURE);
133  }
134  s += param_load(xcsf, fp);
135  s += clset_pset_load(xcsf, fp);
136  fclose(fp);
137  return s;
138 }

165 void
166 xcsf_ae_to_classifier(struct XCSF *xcsf, const int y_dim, const int n_del)
167 {
168  pa_free(xcsf);
169  param_set_y_dim(xcsf, y_dim);
170  param_set_loss_func(xcsf, LOSS_ONEHOT);
171  pa_init(xcsf);
172  const struct Clist *iter = xcsf->pset.list;
173  while (iter != NULL) {
174  free(iter->cl->prediction);
175  iter->cl->prediction = calloc(xcsf->y_dim, sizeof(double));
176  pred_neural_ae_to_classifier(xcsf, iter->cl, n_del);
177  iter->cl->fit = xcsf->INIT_FITNESS;
178  iter->cl->err = xcsf->INIT_ERROR;
179  iter->cl->exp = 0;
180  iter->cl->time = xcsf->time;
181  iter = iter->next;
182  }
183 }

189 void
190 xcsf_store_pset(struct XCSF *xcsf)
191 {
192  clset_kill(xcsf, &xcsf->prev_pset);
193  const struct Clist *iter = xcsf->pset.list;
194  while (iter != NULL) {
195  struct Cl *new = malloc(sizeof(struct Cl));
196  const struct Cl *src = iter->cl;
197  cl_init_copy(xcsf, new, src);
198  clset_add(&xcsf->prev_pset, new);
199  iter = iter->next;
200  }
201 }
