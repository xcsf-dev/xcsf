36 {
37  if (argc < 3 || argc > 5) {
38  printf("Usage: xcsf problemType{csv|mp|maze} ");
39  printf("problem{.csv|size|maze} [config.json] [xcs.bin]\n");
40  exit(EXIT_FAILURE);
41  }
42  struct XCSF *xcsf = malloc(sizeof(struct XCSF));
43  rand_init();
44  env_init(xcsf, argv); // initialise environment and default parameters
45  if (argc > 3) { // load parameter config
46  config_read(xcsf, argv[3]);
47  } else {
48  config_read(xcsf, "default.json");
49  }
50  xcsf_init(xcsf); // initialise XCSF
51  if (argc == 5) { // reload state of a previous experiment
52  const size_t s = xcsf_load(xcsf, argv[4]);
53  printf("XCSF loaded: %d elements\n", (int) s);
54  }
55  param_print(xcsf); // print parameters used
56  if (strcmp(argv[1], "csv") == 0) { // supervised regression - csv file
57  const struct EnvCSV *env = xcsf->env;
58  xcs_supervised_fit(xcsf, env->train_data, env->test_data, true);
59  } else { // reinforcement learning - maze or mux
60  xcs_rl_exp(xcsf);
61  }
62  pa_free(xcsf); // clean up
63  env_free(xcsf);
64  xcsf_free(xcsf);
65  param_free(xcsf);
