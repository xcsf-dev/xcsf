104 char *
105 param_json_export(const struct XCSF *xcsf)
106 {
107  char v[256];
108  snprintf(v, 256, "%d.%d.%d", VERSION_MAJOR, VERSION_MINOR, VERSION_BUILD);
109  cJSON *json = cJSON_CreateObject();
110  cJSON_AddStringToObject(json, "version", v);
111  cJSON_AddNumberToObject(json, "x_dim", xcsf->x_dim);
112  cJSON_AddNumberToObject(json, "y_dim", xcsf->y_dim);
113  cJSON_AddNumberToObject(json, "n_actions", xcsf->n_actions);
114  cJSON_AddNumberToObject(json, "omp_num_threads", xcsf->OMP_NUM_THREADS);
115  cJSON_AddBoolToObject(json, "pop_init", xcsf->POP_INIT);
116  cJSON_AddNumberToObject(json, "max_trials", xcsf->MAX_TRIALS);
117  cJSON_AddNumberToObject(json, "perf_trials", xcsf->PERF_TRIALS);
118  cJSON_AddNumberToObject(json, "pop_size", xcsf->POP_SIZE);
119  cJSON_AddStringToObject(json, "loss_func",
120  loss_type_as_string(xcsf->LOSS_FUNC));
121  if (xcsf->LOSS_FUNC == LOSS_HUBER) {
122  cJSON_AddNumberToObject(json, "huber_delta", xcsf->HUBER_DELTA);
123  }
124  if (xcsf->n_actions > 1) {
125  cJSON_AddNumberToObject(json, "gamma", xcsf->GAMMA);
126  cJSON_AddNumberToObject(json, "teletransportation",
127  xcsf->TELETRANSPORTATION);
128  cJSON_AddNumberToObject(json, "p_explore", xcsf->P_EXPLORE);
129  }
130  cJSON_AddBoolToObject(json, "set_subsumption", xcsf->SET_SUBSUMPTION);
131  cJSON_AddNumberToObject(json, "theta_sub", xcsf->THETA_SUB);
132  cJSON_AddNumberToObject(json, "e0", xcsf->E0);
133  cJSON_AddNumberToObject(json, "alpha", xcsf->ALPHA);
134  cJSON_AddNumberToObject(json, "nu", xcsf->NU);
135  cJSON_AddNumberToObject(json, "beta", xcsf->BETA);
136  cJSON_AddNumberToObject(json, "delta", xcsf->DELTA);
137  cJSON_AddNumberToObject(json, "theta_del", xcsf->THETA_DEL);
138  cJSON_AddNumberToObject(json, "init_fitness", xcsf->INIT_FITNESS);
139  cJSON_AddNumberToObject(json, "init_error", xcsf->INIT_ERROR);
140  cJSON_AddNumberToObject(json, "m_probation", xcsf->M_PROBATION);
141  cJSON_AddBoolToObject(json, "stateful", xcsf->STATEFUL);
142  cJSON_AddBoolToObject(json, "compaction", xcsf->COMPACTION);
143  cJSON_AddItemToObject(json, "ea", cJSON_Parse(ea_param_json_export(xcsf)));
144  switch (xcsf->cond->type) {
145  case RULE_TYPE_DGP:
146  case RULE_TYPE_NEURAL:
147  case RULE_TYPE_NETWORK:
148  break;
149  default:
150  if (xcsf->n_actions > 1) {
151  cJSON *act_params = cJSON_Parse(action_param_json_export(xcsf));
152  cJSON_AddItemToObject(json, "action", act_params);
153  }
154  break;
155  }
156  cJSON *cond_params = cJSON_Parse(cond_param_json_export(xcsf));
157  cJSON_AddItemToObject(json, "condition", cond_params);
158  cJSON *pred_params = cJSON_Parse(pred_param_json_export(xcsf));
159  cJSON_AddItemToObject(json, "prediction", pred_params);
160  char *string = cJSON_Print(json);
161  cJSON_Delete(json);
162  return string;
163 }

304 void
305 param_json_import(struct XCSF *xcsf, const char *json_str)
306 {
307  cJSON *json = cJSON_Parse(json_str);
308  if (json == NULL) {
309  printf("Error reading JSON\n");
310  const char *error_ptr = cJSON_GetErrorPtr();
311  if (error_ptr != NULL) {
312  printf("Error before: %s\n", error_ptr);
313  }
314  exit(EXIT_FAILURE);
315  }
316  for (cJSON *iter = json->child; iter != NULL; iter = iter->next) {
317  if (param_json_import_general(xcsf, iter)) {
318  continue;
319  }
320  if (param_json_import_multi(xcsf, iter)) {
321  continue;
322  }
323  if (param_json_import_subsump(xcsf, iter)) {
324  continue;
325  }
326  if (param_json_import_cl_general(xcsf, iter)) {
327  continue;
328  }
329  if (strncmp(iter->string, "ea\0", 3) == 0) {
330  ea_param_json_import(xcsf, iter->child);
331  continue;
332  }
333  if (strncmp(iter->string, "action\0", 7) == 0) {
334  action_param_json_import(xcsf, iter->child);
335  continue;
336  }
337  if (strncmp(iter->string, "condition\0", 10) == 0) {
338  cond_param_json_import(xcsf, iter->child);
339  continue;
340  }
341  if (strncmp(iter->string, "prediction\0", 11) == 0) {
342  pred_param_json_import(xcsf, iter->child);
343  continue;
344  }
345  printf("Error: unable to import parameter: %s\n", iter->string);
346  exit(EXIT_FAILURE);
347  }
348  cJSON_Delete(json);
349 }
