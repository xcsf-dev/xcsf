44 void
45 param_init(struct XCSF *xcsf, const int x_dim, const int y_dim,
46  const int n_actions)
47 {
48  xcsf->time = 0;
49  xcsf->error = xcsf->E0;
50  xcsf->mset_size = 0;
51  xcsf->aset_size = 0;
52  xcsf->mfrac = 0;
53  xcsf->ea = malloc(sizeof(struct ArgsEA));
54  xcsf->act = malloc(sizeof(struct ArgsAct));
55  xcsf->cond = malloc(sizeof(struct ArgsCond));
56  xcsf->pred = malloc(sizeof(struct ArgsPred));
57  xcsf->population_file = malloc(sizeof(char));
58  xcsf->population_file[0] = '\0';
59  param_set_n_actions(xcsf, n_actions);
60  param_set_x_dim(xcsf, x_dim);
61  param_set_y_dim(xcsf, y_dim);
62  param_set_omp_num_threads(xcsf, 8);
63  param_set_random_state(xcsf, 0);
64  param_set_pop_init(xcsf, true);
65  param_set_max_trials(xcsf, 100000);
66  param_set_perf_trials(xcsf, 1000);
67  param_set_pop_size(xcsf, 2000);
68  param_set_loss_func(xcsf, LOSS_MAE);
69  param_set_huber_delta(xcsf, 1);
70  param_set_e0(xcsf, 0.01);
71  param_set_alpha(xcsf, 0.1);
72  param_set_nu(xcsf, 5);
73  param_set_beta(xcsf, 0.1);
74  param_set_delta(xcsf, 0.1);
75  param_set_theta_del(xcsf, 20);
76  param_set_init_fitness(xcsf, 0.01);
77  param_set_init_error(xcsf, 0);
78  param_set_m_probation(xcsf, 10000);
79  param_set_stateful(xcsf, true);
80  param_set_compaction(xcsf, false);
81  param_set_gamma(xcsf, 0.95);
82  param_set_teletransportation(xcsf, 50);
83  param_set_p_explore(xcsf, 0.9);
84  param_set_set_subsumption(xcsf, false);
85  param_set_theta_sub(xcsf, 100);
86  ea_param_defaults(xcsf);
87  action_param_defaults(xcsf);
88  cond_param_defaults(xcsf);
89  pred_param_defaults(xcsf);

112 char *
113 param_json_export(const struct XCSF *xcsf)
114 {
115  char v[256];
116  snprintf(v, 256, "%d.%d.%d", VERSION_MAJOR, VERSION_MINOR, VERSION_BUILD);
117  cJSON *json = cJSON_CreateObject();
118  cJSON_AddStringToObject(json, "version", v);
119  cJSON_AddNumberToObject(json, "x_dim", xcsf->x_dim);
120  cJSON_AddNumberToObject(json, "y_dim", xcsf->y_dim);
121  cJSON_AddNumberToObject(json, "n_actions", xcsf->n_actions);
122  cJSON_AddNumberToObject(json, "omp_num_threads", xcsf->OMP_NUM_THREADS);
123  cJSON_AddNumberToObject(json, "random_state", xcsf->RANDOM_STATE);
124  if (xcsf->population_file != NULL) {
125  cJSON_AddStringToObject(json, "population_file", xcsf->population_file);
126  }
127  cJSON_AddBoolToObject(json, "pop_init", xcsf->POP_INIT);
128  cJSON_AddNumberToObject(json, "max_trials", xcsf->MAX_TRIALS);
129  cJSON_AddNumberToObject(json, "perf_trials", xcsf->PERF_TRIALS);
130  cJSON_AddNumberToObject(json, "pop_size", xcsf->POP_SIZE);
131  cJSON_AddStringToObject(json, "loss_func",
132  loss_type_as_string(xcsf->LOSS_FUNC));
133  if (xcsf->LOSS_FUNC == LOSS_HUBER) {
134  cJSON_AddNumberToObject(json, "huber_delta", xcsf->HUBER_DELTA);
135  }
136  if (xcsf->n_actions > 1) {
137  cJSON_AddNumberToObject(json, "gamma", xcsf->GAMMA);
138  cJSON_AddNumberToObject(json, "teletransportation",
139  xcsf->TELETRANSPORTATION);
140  cJSON_AddNumberToObject(json, "p_explore", xcsf->P_EXPLORE);
141  }
142  cJSON_AddBoolToObject(json, "set_subsumption", xcsf->SET_SUBSUMPTION);
143  cJSON_AddNumberToObject(json, "theta_sub", xcsf->THETA_SUB);
144  cJSON_AddNumberToObject(json, "e0", xcsf->E0);
145  cJSON_AddNumberToObject(json, "alpha", xcsf->ALPHA);
146  cJSON_AddNumberToObject(json, "nu", xcsf->NU);
147  cJSON_AddNumberToObject(json, "beta", xcsf->BETA);
148  cJSON_AddNumberToObject(json, "delta", xcsf->DELTA);
149  cJSON_AddNumberToObject(json, "theta_del", xcsf->THETA_DEL);
150  cJSON_AddNumberToObject(json, "init_fitness", xcsf->INIT_FITNESS);
151  cJSON_AddNumberToObject(json, "init_error", xcsf->INIT_ERROR);
152  cJSON_AddNumberToObject(json, "m_probation", xcsf->M_PROBATION);
153  cJSON_AddBoolToObject(json, "stateful", xcsf->STATEFUL);
154  cJSON_AddBoolToObject(json, "compaction", xcsf->COMPACTION);
155  char *ea_param_str = ea_param_json_export(xcsf);
156  cJSON *ea_params = cJSON_Parse(ea_param_str);
157  cJSON_AddItemToObject(json, "ea", ea_params);
158  free(ea_param_str);
159  switch (xcsf->cond->type) {
160  case RULE_TYPE_DGP:
161  case RULE_TYPE_NEURAL:
162  case RULE_TYPE_NETWORK:
163  break;
164  default:
165  if (xcsf->n_actions > 1) {
166  char *act_param_str = action_param_json_export(xcsf);
167  cJSON *act_params = cJSON_Parse(act_param_str);
168  cJSON_AddItemToObject(json, "action", act_params);
169  free(act_param_str);
170  }
171  break;
172  }
173  char *cond_param_str = cond_param_json_export(xcsf);
174  cJSON *cond_params = cJSON_Parse(cond_param_str);
175  cJSON_AddItemToObject(json, "condition", cond_params);
176  free(cond_param_str);
177  char *pred_param_str = pred_param_json_export(xcsf);
178  cJSON *pred_params = cJSON_Parse(pred_param_str);
179  cJSON_AddItemToObject(json, "prediction", pred_params);
180  free(pred_param_str);
181  char *string = cJSON_Print(json);
182  cJSON_Delete(json);
183  return string;
184 }

209  } else if (strncmp(json->string, "random_state\0", 13) == 0 &&

421 void
422 param_json_import(struct XCSF *xcsf, const char *json_str)
423 {
424  cJSON *json = cJSON_Parse(json_str);
425  utils_json_parse_check(json);
426  for (cJSON *iter = json->child; iter != NULL; iter = iter->next) {
427  if (param_json_import_general(xcsf, iter)) {
428  continue;
429  }
430  if (param_json_import_multi(xcsf, iter)) {
431  continue;
432  }
433  if (param_json_import_subsump(xcsf, iter)) {
434  continue;
435  }
436  if (param_json_import_cl_general(xcsf, iter)) {
437  continue;
438  }
439  if (strncmp(iter->string, "ea\0", 3) == 0) {
440  ea_param_json_import(xcsf, iter->child);
441  continue;
442  }
443  if (strncmp(iter->string, "action\0", 7) == 0) {
444  param_json_import_action(xcsf, iter->child);
445  continue;
446  }
447  if (strncmp(iter->string, "condition\0", 10) == 0) {
448  param_json_import_condition(xcsf, iter->child);
449  continue;
450  }
451  if (strncmp(iter->string, "prediction\0", 11) == 0) {
452  param_json_import_prediction(xcsf, iter->child);
453  continue;
454  }
455  printf("Error: unable to import parameter: %s\n", iter->string);
456  exit(EXIT_FAILURE);
457  }
458  cJSON_Delete(json);
459 }

532 size_t
533 param_load(struct XCSF *xcsf, FILE *fp)
534 {
535  size_t s = 0;
536  size_t len = 0;
537  s += fread(&len, sizeof(size_t), 1, fp);
538  free(xcsf->population_file);
539  xcsf->population_file = malloc(sizeof(char) * len);
540  s += fread(xcsf->population_file, sizeof(char), len, fp);
541  s += fread(&xcsf->time, sizeof(int), 1, fp);
542  s += fread(&xcsf->error, sizeof(double), 1, fp);
543  s += fread(&xcsf->mset_size, sizeof(double), 1, fp);
544  s += fread(&xcsf->aset_size, sizeof(double), 1, fp);
545  s += fread(&xcsf->mfrac, sizeof(double), 1, fp);
546  s += fread(&xcsf->explore, sizeof(bool), 1, fp);
547  s += fread(&xcsf->x_dim, sizeof(int), 1, fp);
548  s += fread(&xcsf->y_dim, sizeof(int), 1, fp);
549  s += fread(&xcsf->n_actions, sizeof(int), 1, fp);
550  if (xcsf->x_dim < 1 || xcsf->y_dim < 1 || xcsf->n_actions < 1) {
551  printf("param_load(): read error\n");
552  exit(EXIT_FAILURE);
553  }
554  s += fread(&xcsf->OMP_NUM_THREADS, sizeof(int), 1, fp);
555  s += fread(&xcsf->RANDOM_STATE, sizeof(int), 1, fp);
556  s += fread(&xcsf->POP_INIT, sizeof(bool), 1, fp);
557  s += fread(&xcsf->MAX_TRIALS, sizeof(int), 1, fp);
558  s += fread(&xcsf->PERF_TRIALS, sizeof(int), 1, fp);
559  s += fread(&xcsf->POP_SIZE, sizeof(int), 1, fp);
560  s += fread(&xcsf->LOSS_FUNC, sizeof(int), 1, fp);
561  s += fread(&xcsf->HUBER_DELTA, sizeof(double), 1, fp);
562  s += fread(&xcsf->GAMMA, sizeof(double), 1, fp);
563  s += fread(&xcsf->TELETRANSPORTATION, sizeof(int), 1, fp);
564  s += fread(&xcsf->P_EXPLORE, sizeof(double), 1, fp);
565  s += fread(&xcsf->SET_SUBSUMPTION, sizeof(bool), 1, fp);
566  s += fread(&xcsf->THETA_SUB, sizeof(int), 1, fp);
