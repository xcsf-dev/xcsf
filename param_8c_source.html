316 void
317 param_init(struct XCSF *xcsf, const int x_dim, const int y_dim,
318  const int n_actions)
319 {
320  xcsf->time = 0;
321  xcsf->error = xcsf->E0;
322  xcsf->mset_size = 0;
323  xcsf->aset_size = 0;
324  xcsf->mfrac = 0;
325  xcsf->ea = malloc(sizeof(struct ArgsEA));
326  xcsf->act = malloc(sizeof(struct ArgsAct));
327  xcsf->cond = malloc(sizeof(struct ArgsCond));
328  xcsf->pred = malloc(sizeof(struct ArgsPred));
329  param_set_n_actions(xcsf, n_actions);
330  param_set_x_dim(xcsf, x_dim);
331  param_set_y_dim(xcsf, y_dim);
332  param_defaults_cl_general(xcsf);
333  param_defaults_general(xcsf);
334  param_defaults_multistep(xcsf);
335  param_defaults_subsumption(xcsf);
336  ea_param_defaults(xcsf);
337  action_param_defaults(xcsf);
338  cond_param_defaults(xcsf);
339  pred_param_defaults(xcsf);
340 }

358 void
359 param_print(const struct XCSF *xcsf)
360 {
361  printf("VERSION=%d.%d.%d, ", VERSION_MAJOR, VERSION_MINOR, VERSION_BUILD);
362  param_print_general(xcsf);
363  if (xcsf->n_actions > 1) {
364  param_print_multistep(xcsf);
365  }
366  param_print_subsumption(xcsf);
367  param_print_cl_general(xcsf);
368  ea_param_print(xcsf);
369  switch (xcsf->cond->type) {
370  case RULE_TYPE_DGP:
371  case RULE_TYPE_NEURAL:
372  case RULE_TYPE_NETWORK:
373  break;
374  default:
375  if (xcsf->n_actions > 1) {
376  action_param_print(xcsf);
377  }
378  break;
379  }
380  cond_param_print(xcsf);
381  pred_param_print(xcsf);
382  printf("\n");
383 }

421 size_t
422 param_load(struct XCSF *xcsf, FILE *fp)
423 {
424  size_t s = 0;
425  s += fread(&xcsf->time, sizeof(int), 1, fp);
426  s += fread(&xcsf->error, sizeof(double), 1, fp);
427  s += fread(&xcsf->mset_size, sizeof(double), 1, fp);
428  s += fread(&xcsf->aset_size, sizeof(double), 1, fp);
429  s += fread(&xcsf->mfrac, sizeof(double), 1, fp);
430  s += fread(&xcsf->explore, sizeof(bool), 1, fp);
431  s += fread(&xcsf->x_dim, sizeof(int), 1, fp);
432  s += fread(&xcsf->y_dim, sizeof(int), 1, fp);
433  s += fread(&xcsf->n_actions, sizeof(int), 1, fp);
434  if (xcsf->x_dim < 1 || xcsf->y_dim < 1) {
435  printf("param_load(): read error\n");
436  exit(EXIT_FAILURE);
437  }
438  s += param_load_general(xcsf, fp);
439  s += param_load_multistep(xcsf, fp);
440  s += param_load_subsumption(xcsf, fp);
441  s += param_load_cl_general(xcsf, fp);
442  s += ea_param_load(xcsf, fp);
443  s += action_param_load(xcsf, fp);
444  s += cond_param_load(xcsf, fp);
445  s += pred_param_load(xcsf, fp);
446  return s;
447 }
