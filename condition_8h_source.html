302 static inline void
303 cond_json_import(const struct XCSF *xcsf, struct Cl *c, const cJSON *json)
304 {
305  const cJSON *item = cJSON_GetObjectItem(json, "type");
306  if (item == NULL || !cJSON_IsString(item)) {
307  printf("cond_json_import(): missing type\n");
308  exit(EXIT_FAILURE);
309  }
310  const char *type = item->valuestring;
311  if (condition_type_as_int(type) != xcsf->cond->type) {
312  printf("cond_json_import(): mismatched type\n");
313  printf("XCSF type = %s, but imported type = %s\n",
314  condition_type_as_string(xcsf->cond->type), type);
315  exit(EXIT_FAILURE);
316  }
317  (*c->cond_vptr->cond_impl_json_import)(xcsf, c, json);
318 }
