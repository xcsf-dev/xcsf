112bool
113cond_dgp_match(const struct XCSF *xcsf, const struct Cl *c, const double *x)
114{
115 const struct CondDGP *cond = c->cond;
116 graph_update(&cond->dgp, x, !xcsf->STATEFUL);
117 if (graph_output(&cond->dgp, 0) > 0.5) {
118 return true;
119 }
120 return false;
121}
