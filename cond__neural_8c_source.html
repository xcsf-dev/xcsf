243 int
244 cond_neural_neurons(const struct XCSF *xcsf, const struct Cl *c, int layer)
245 {
246  (void) xcsf;
247  const struct CondNeural *cond = c->cond;
248  const struct Net *net = &cond->net;
249  const struct Llist *iter = net->tail;
250  int i = 0;
251  while (iter != NULL) {
252  if (i == layer) {
253  if (iter->layer->type == CONVOLUTIONAL) {
254  return iter->layer->n_filters;
255  }
256  return iter->layer->n_outputs;
257  }
258  iter = iter->prev;
259  ++i;
260  }
261  return 0;
262 }
