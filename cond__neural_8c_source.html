244 int
245 cond_neural_neurons(const struct XCSF *xcsf, const struct Cl *c, int layer)
246 {
247  (void) xcsf;
248  const struct CondNeural *cond = c->cond;
249  const struct Net *net = &cond->net;
250  const struct Llist *iter = net->tail;
251  int i = 0;
252  while (iter != NULL) {
253  if (i == layer) {
254  if (iter->layer->type == CONVOLUTIONAL) {
255  return iter->layer->n_filters;
256  }
257  return iter->layer->n_outputs;
258  }
259  iter = iter->prev;
260  ++i;
261  }
262  return 0;
263 }

350 cond_neural_param_json_import(struct XCSF *xcsf, cJSON *json)
351 {
352  layer_args_free(&xcsf->cond->largs);
353  for (cJSON *iter = json; iter != NULL; iter = iter->next) {
354  struct ArgsLayer *larg = malloc(sizeof(struct ArgsLayer));
355  layer_args_init(larg);
356  larg->n_inputs = xcsf->x_dim;
357  layer_args_json_import(larg, iter->child);
358  if (xcsf->cond->largs == NULL) {
359  xcsf->cond->largs = larg;
360  } else {
361  struct ArgsLayer *layer_iter = xcsf->cond->largs;
362  while (layer_iter->next != NULL) {
363  layer_iter = layer_iter->next;
364  }
365  layer_iter->next = larg;
366  }
367  }
368  layer_args_validate(xcsf->cond->largs);
369 }
