242 int
243 cond_neural_neurons(const struct XCSF *xcsf, const struct Cl *c, int layer)
244 {
245  (void) xcsf;
246  const struct CondNeural *cond = c->cond;
247  const struct Net *net = &cond->net;
248  const struct Llist *iter = net->tail;
249  int i = 0;
250  while (iter != NULL) {
251  if (i == layer) {
252  if (iter->layer->type == CONVOLUTIONAL) {
253  return iter->layer->n_filters;
254  }
255  return iter->layer->n_outputs;
256  }
257  iter = iter->prev;
258  ++i;
259  }
260  return 0;
261 }
