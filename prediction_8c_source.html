119 static void
120 pred_param_defaults_neural(struct XCSF *xcsf)
121 {
122  // hidden layer
123  struct ArgsLayer *la = malloc(sizeof(struct ArgsLayer));
124  layer_args_init(la);
125  la->type = CONNECTED;
126  la->n_inputs = xcsf->x_dim;
127  la->n_init = 10;
128  la->n_max = 100;
129  la->max_neuron_grow = 1;
130  la->function = LOGISTIC;
131  la->evolve_weights = true;
132  la->evolve_neurons = true;
133  la->evolve_connect = true;
134  la->evolve_eta = true;
135  la->sgd_weights = true;
136  la->eta = 0.01;
137  la->momentum = 0.9;
138  xcsf->pred->largs = la;
139  // output layer
140  la->next = layer_args_copy(la);
141  la->next->n_inputs = la->n_init;
142  la->next->n_init = xcsf->y_dim;
143  la->next->n_max = xcsf->y_dim;
144  la->next->evolve_neurons = false;
145 }

296 void
297 pred_transform_input(const struct XCSF *xcsf, const double *x, const double X0,
298  double *tmp_input)
299 {
300  // bias term
301  tmp_input[0] = X0;
302  int idx = 1;
303  // linear terms
304  for (int i = 0; i < xcsf->x_dim; ++i) {
305  tmp_input[idx] = x[i];
306  ++idx;
307  }
308  // quadratic terms
309  if (xcsf->pred->type == PRED_TYPE_NLMS_QUADRATIC ||
310  xcsf->pred->type == PRED_TYPE_RLS_QUADRATIC) {
311  for (int i = 0; i < xcsf->x_dim; ++i) {
312  for (int j = i; j < xcsf->x_dim; ++j) {
313  tmp_input[idx] = x[i] * x[j];
314  ++idx;
315  }
316  }
317  }
318 }
