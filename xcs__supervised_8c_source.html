56 static void
57 xcs_supervised_trial(struct XCSF *xcsf, const double *x, const double *y)
58 {
59  clset_init(&xcsf->mset);
60  clset_init(&xcsf->kset);
61  clset_match(xcsf, x);
62  pa_build(xcsf, x);
63  if (xcsf->explore) {
64  clset_update(xcsf, &xcsf->mset, x, y, true);
65  ea(xcsf, &xcsf->mset);
66  }
67  clset_kill(xcsf, &xcsf->kset);
68  clset_free(&xcsf->mset);
69 }

80 double
81 xcs_supervised_fit(struct XCSF *xcsf, const struct Input *train_data,
82  const struct Input *test_data, const bool shuffle)
83 {
84  if (xcsf->time == 0) {
85  clset_pset_init(xcsf);
86  }
87  double err = 0; // training error: total over all trials
88  double werr = 0; // training error: windowed total
89  double wterr = 0; // testing error: windowed total
90  for (int cnt = 0; cnt < xcsf->MAX_TRIALS; ++cnt) {
91  // training sample
92  int row = xcs_supervised_sample(train_data, cnt, shuffle);
93  const double *x = &train_data->x[row * train_data->x_dim];
94  const double *y = &train_data->y[row * train_data->y_dim];
95  param_set_explore(xcsf, true);
96  xcs_supervised_trial(xcsf, x, y);
97  const double error = (xcsf->loss_ptr)(xcsf, xcsf->pa, y);
98  werr += error;
99  err += error;
100  xcsf->error += (error - xcsf->error) * xcsf->BETA;
101  // test sample
102  if (test_data != NULL) {
103  row = xcs_supervised_sample(test_data, cnt, shuffle);
104  x = &test_data->x[row * test_data->x_dim];
105  y = &test_data->y[row * test_data->y_dim];
106  param_set_explore(xcsf, false);
107  xcs_supervised_trial(xcsf, x, y);
108  wterr += (xcsf->loss_ptr)(xcsf, xcsf->pa, y);
109  }
110  perf_print(xcsf, &werr, &wterr, cnt);
111  }
112  return err / xcsf->MAX_TRIALS;
113 }

175  }
